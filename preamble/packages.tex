% packages.tex
% Gói lệnh sử dụng chung cho sách

% ---- Core and Language ----
\usepackage{fontspec}      % For loading modern fonts (requires XeLaTeX/LuaTeX)
\usepackage{polyglossia}   % Modern multilingual support

% ---- Mathematics ----
\usepackage{amsmath}
\usepackage{amssymb}
% amsthm is no longer needed, as tcolorbox will handle theorem environments

% ---- Graphics, Layout, and Color ----
\usepackage{graphicx}
\usepackage[a4paper, margin=2.5cm, headheight=15pt]{geometry} % Adjusted headheight for fancyhdr
\usepackage{fancyhdr}
\usepackage{xcolor}
\usepackage{fontawesome5}  % For icons (with XeLaTeX compatibility)

% ---- Advanced Boxes and Theorems ----
\usepackage[most]{tcolorbox}
\tcbuselibrary{theorems}    % Essential for \newtcbtheorem
\tcbuselibrary{skins}       % For advanced styling like 'enhanced'
\tcbuselibrary{breakable}   % Allows boxes to break across pages
\usepackage{xparse}         % For creating advanced commands/environments

% ---- Headings and Table of Contents ----
\usepackage{titlesec}
\usepackage{titletoc}

% ---- Utilities ----
\usepackage{enumitem}      % For better lists
\usepackage[colorlinks=true, allcolors=myblue]{hyperref} % Using our custom color
